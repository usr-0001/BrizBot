class HostDB:
    def __init__(self):
        pass

    def run_query(self,  query: str, commit: bool = False):
        with mysql.connect(
            host=app_settings.dictionary['database']['host'],
            database=app_settings.dictionary['database']['database'],
            user=app_settings.dictionary['database']['user'],
            password=app_settings.dictionary['database']['password'],
            port=3306
        ) as conn:
            with conn.cursor() as cursor:
                cursor.execute(query)
            if commit:
                conn.commit()
            conn.close()
            return cursor
